

CREATE TABLE persona (
    id_persona INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    apellidos VARCHAR(150),
    email VARCHAR(150) UNIQUE,
    telefono VARCHAR(20)
);

CREATE TABLE usuario (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    id_persona INT,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    rol ENUM('admin','empleado','cliente')
);

CREATE TABLE empleado (
    id_empleado INT AUTO_INCREMENT PRIMARY KEY,
    id_persona INT,
    nif VARCHAR(15) UNIQUE,
    cuenta VARCHAR(34),
    departamento VARCHAR(100),
    salario DECIMAL(10,2),
    puesto VARCHAR(100)
);

CREATE TABLE cliente (
    id_cliente INT AUTO_INCREMENT PRIMARY KEY,
    id_persona INT,
    fecha_alta DATE
);

CREATE TABLE campania (
    id_campania INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    descripcion TEXT,
    fecha_inicio DATE,
    fecha_fin DATE
);

CREATE TABLE campania_cliente (
    id_campania INT,
    id_cliente INT,
    PRIMARY KEY (id_campania, id_cliente)
);

CREATE TABLE atributo_cliente (
    id_atributo INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    texto TEXT,
    contacto VARCHAR(100)
);

CREATE TABLE atributo_cliente_cliente (
    id_cliente INT,
    id_atributo INT,
    valor VARCHAR(255),
    PRIMARY KEY (id_cliente, id_atributo)
);

CREATE TABLE proveedor (
    id_proveedor INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(150),
    cif VARCHAR(15) UNIQUE,
    direccion VARCHAR(200),
    telefono VARCHAR(20),
    contacto VARCHAR(100)
);

CREATE TABLE producto (
    id_producto INT AUTO_INCREMENT PRIMARY KEY,
    id_proveedor INT,
    nombre VARCHAR(150),
    precio DECIMAL(8,2),
    cantidad INT,
    tipo ENUM('ingrediente','otros')
);

CREATE TABLE plato (
    id_plato INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(150),
    descripcion TEXT,
    precio DECIMAL(8,2)
);

CREATE TABLE plato_producto (
    id_plato INT,
    id_producto INT,
    cantidad DECIMAL(6,2),
    PRIMARY KEY (id_plato, id_producto)
);

CREATE TABLE reserva (
    id_reserva INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT,
    fecha DATE,
    hora TIME,
    personas INT,
    tipo_promocion VARCHAR(100),
    estado ENUM('Confirmado','Cancelado')
);

CREATE TABLE pedido (
    id_pedido INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente INT,
    id_reserva INT,
    fecha DATE,
    hora TIME,
    total DECIMAL(10,2)
);

CREATE TABLE pedido_plato (
    id_pedido INT,
    id_plato INT,
    cantidad INT,
    PRIMARY KEY (id_pedido, id_plato)
);

CREATE TABLE factura (
    id_factura INT AUTO_INCREMENT PRIMARY KEY,
    id_pedido INT,
    fecha DATE,
    total DECIMAL(10,2),
    pagada ENUM('Si','No'),
    cod_promocion VARCHAR(50)
);
